# Initial Django Setup
python -m pip install Django  # Install Django
django-admin startproject Greaves  # Create new Django project
cd Greaves  # Change into project directory
python manage.py startapp LEU1  # Create new Django app
python manage.py makemigrations  # Create database migrations
python manage.py migrate  # Apply database migrations
python manage.py createsuperuser  # Create admin user
# Remember to add app name to INSTALLED_APPS in settings.py

# Database Migration Commands
python manage.py inspectdb > models.py  # Generate models from existing database
python manage.py inspectdb --database=default > ATP/models3.py
python manage.py inspectdb --database=leu1_atp > LEU1_ATP/models3.py 
python manage.py inspectdb --database=leu1_mc_shop > Mc_Shop/models3.py
python manage.py migrate --fake ATP zero  # Fake migration for ATP app
python manage.py makemigrations Mc_Shop  # Create migrations for Mc_Shop app

# Project Structure
pip install django-extensions  # Install django-extensions
